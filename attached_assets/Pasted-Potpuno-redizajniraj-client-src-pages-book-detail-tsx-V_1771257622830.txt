Potpuno redizajniraj client/src/pages/book-detail.tsx:

VAŽNO: Fokus je na FIZIČKIM knjigama, NE online čitanju!
PDF linkovi su samo za UZORKE knjiga (prvih par stranica).

ZAMIJENI CIJELI SADRŽAJ book-detail.tsx fajla sa ovim kodom:

import { useQuery } from "@tanstack/react-query";
import { useParams, Link } from "wouter";
import { Card, CardContent, CardHeader, CardTitle, CardDescription } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Badge } from "@/components/ui/badge";
import { Alert, AlertTitle, AlertDescription } from "@/components/ui/alert";
import { 
  BookOpen, 
  ShoppingCart, 
  Brain, 
  MapPin, 
  Library,
  Eye,
  Info,
  ArrowLeft
} from "lucide-react";
import { useAuth } from "@/hooks/use-auth";

export default function BookDetailPage() {
  const { id } = useParams();
  const { user } = useAuth();
  
  const { data: book, isLoading } = useQuery({
    queryKey: [`/api/books/${id}`],
  });

  const { data: quizzes } = useQuery({
    queryKey: [`/api/books/${id}/quizzes`],
    enabled: !!book,
  });

  if (isLoading) {
    return <div className="flex justify-center p-8">Učitavanje...</div>;
  }

  if (!book) {
    return <div className="p-8">Knjiga nije pronađena</div>;
  }

  const quiz = quizzes?.[0];

  return (
    <div className="container mx-auto px-4 py-8 max-w-4xl">
      {/* Back button */}
      <Link href={user ? "/ucenik/biblioteka" : "/biblioteka"}>
        <Button variant="ghost" className="mb-4">
          <ArrowLeft className="mr-2 h-4 w-4" />
          Nazad na biblioteku
        </Button>
      </Link>

      {/* Hero sekcija sa naslovnicom */}
      <div className="grid md:grid-cols-[300px_1fr] gap-8 mb-8">
        <div>
          <img
            src={book.coverImage || "/placeholder-book.jpg"}
            alt={book.title}
            className="w-full rounded-lg shadow-lg"
          />
        </div>
        <div>
          <h1 className="text-4xl font-bold mb-2">{book.title}</h1>
          <p className="text-xl text-muted-foreground mb-4">{book.author}</p>
          <div className="flex flex-wrap gap-2 mb-4">
            {book.ageGroup && <Badge>{book.ageGroup}</Badge>}
            {book.genre && <Badge variant="outline">{book.genre}</Badge>}
            {book.readingDifficulty && <Badge variant="secondary">{book.readingDifficulty}</Badge>}
          </div>
          {book.description && <p className="text-muted-foreground">{book.description}</p>}
        </div>
      </div>

      {/* GDE NAĆI KNJIGU - GLAVNI FOKUS */}
      <Card className="mb-8 border-2 border-primary">
        <CardHeader>
          <CardTitle className="flex items-center gap-2">
            <MapPin className="text-primary" />
            Gdje pronaći ovu knjigu?
          </CardTitle>
        </CardHeader>
        <CardContent className="space-y-4">
          
          {/* Školska biblioteka */}
          {book.availableInLibrary && (
            <Alert>
              <Library className="h-5 w-5" />
              <AlertTitle>Dostupno u školskoj biblioteci</AlertTitle>
              <AlertDescription>
                {book.locationInLibrary && (
                  <p className="mt-2">
                    <strong>Lokacija:</strong> {book.locationInLibrary}
                  </p>
                )}
                {book.copiesAvailable !== undefined && book.copiesAvailable !== null && (
                  <p>
                    <strong>Dostupno primjeraka:</strong> {book.copiesAvailable}
                  </p>
                )}
              </AlertDescription>
            </Alert>
          )}

          {/* Kupovina */}
          {book.purchaseUrl && (
            <Button 
              size="lg" 
              className="w-full" 
              asChild
            >
              <a href={book.purchaseUrl} target="_blank" rel="noopener noreferrer">
                <ShoppingCart className="mr-2" />
                Kupi knjigu online
              </a>
            </Button>
          )}

          {/* PDF uzorak */}
          {book.pdfUrl && (
            <div>
              <p className="text-sm text-muted-foreground mb-2">
                Možeš pogledati prvih nekoliko stranica:
              </p>
              <Button variant="outline" className="w-full" asChild>
                <a href={book.pdfUrl} target="_blank" rel="noopener noreferrer">
                  <Eye className="mr-2" />
                  Pogledaj uzorak knjige (PDF)
                </a>
              </Button>
            </div>
          )}

          {!book.availableInLibrary && !book.purchaseUrl && !book.pdfUrl && (
            <p className="text-sm text-muted-foreground text-center py-4">
              Informacije o dostupnosti knjige nisu trenutno dostupne.
            </p>
          )}
        </CardContent>
      </Card>

      {/* KVIZ SEKCIJA - Tek nakon čitanja */}
      {user && quiz && (
        <Card className="mb-8">
          <CardHeader>
            <CardTitle>Pročitao si knjigu?</CardTitle>
            <CardDescription>
              Dokaži svoje znanje i osvoji bodove!
            </CardDescription>
          </CardHeader>
          <CardContent>
            <Link href={`/ucenik/kviz/${quiz.id}`}>
              <Button size="lg" className="w-full">
                <Brain className="mr-2" />
                Rješavaj kviz
              </Button>
            </Link>
          </CardContent>
        </Card>
      )}

      {!user && (
        <Alert className="mb-8">
          <Info />
          <AlertTitle>Prijavi se za rješavanje kviza</AlertTitle>
          <AlertDescription>
            Moraš biti prijavljen da bi mogao rješavati kvizove.
            <Link href="/prijava">
              <Button variant="link" className="px-0 h-auto ml-1">
                Prijavi se
              </Button>
            </Link>
          </AlertDescription>
        </Alert>
      )}

      {/* DETALJI O KNJIGI */}
      <Card>
        <CardHeader>
          <CardTitle>Detalji o knjizi</CardTitle>
        </CardHeader>
        <CardContent>
          <div className="grid grid-cols-1 md:grid-cols-2 gap-4 text-sm">
            <div>
              <strong>Autor:</strong> {book.author}
            </div>
            {book.pageCount && (
              <div>
                <strong>Broj stranica:</strong> {book.pageCount}
              </div>
            )}
            {book.publisher && (
              <div>
                <strong>Izdavač:</strong> {book.publisher}
              </div>
            )}
            {book.publicationYear && (
              <div>
                <strong>Godina izdanja:</strong> {book.publicationYear}
              </div>
            )}
            {book.isbn && (
              <div>
                <strong>ISBN:</strong> {book.isbn}
              </div>
            )}
            {book.readingDifficulty && (
              <div>
                <strong>Težina čitanja:</strong> {book.readingDifficulty}
              </div>
            )}
            {book.language && (
              <div>
                <strong>Jezik:</strong> {book.language}
              </div>
            )}
            {book.bookFormat && (
              <div>
                <strong>Format:</strong> {book.bookFormat}
              </div>
            )}
            {book.recommendedForGrades && book.recommendedForGrades.length > 0 && (
              <div className="md:col-span-2">
                <strong>Preporučeno za:</strong> {book.recommendedForGrades.join(', ')} razred
              </div>
            )}
          </div>
        </CardContent>
      </Card>
    </div>
  );
}

Javi kad završiš i testiraj da li stranica radi - otvori bilo koju knjigu i provjeri prikaz.
